```javascript Node.js
import { SwervpayClient } from '@swervpaydev/sdk';

const config = {
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

const swyftpay = new SwervpayClient(config);

await swyftpay.fx.exchange({
    amount: 100,
    currency: 'NGN',
    to: 'USD'
})
```

```go Go
package main

import (
	"context"
	"fmt"
	swervpay "github.com/swerv-ltd/swervpay-go"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	client := swervpay.NewSwervpayClient(&swervpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })

    data := swervpay.FxExchange{
        Amount: 100,
        Currency: "NGN",
        To: "USD"
    }

	resp, err := client.Fx.exchange(ctx, data)

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
