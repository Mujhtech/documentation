<RequestExample>

```bash cURL
curl -X POST https://api.swyftpay.io/api/v1/cards/{id}/withdraw \
  -H "Authorization: Bearer <TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{
    "amount": 1000
  }'
```

```javascript Node.js
import { SwervpayClient } from '@swervpaydev/sdk';

const config = {
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

const swyftpay = new SwervpayClient(config);

await swyftpay.card.withdraw('<CARD_ID>',{
    amount: "user"
})
```

```php PHP

$config = [
	'business_id' => '<BUSINESS_ID>',
	'secret_key' => '<SECRET_KEY>'
];

$swyftpayClient = new Swervpay($config);

$swyftpayClient->card->withdraw('<CARD_ID>', [
    "amount" => "user"

```


```go Go
package main

import (
	"context"
	"fmt"
	swyftpay "github.com/Swerv-Ltd/swyftpay-go/lib"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	swyftpayClient := swyftpay.NewAClient(&swyftpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })


    data := swyftpayClient.CardWithdraw{

        Amount: 1000,
    }

	resp, err := swyftpayClient.card.withdraw(ctx, '<CARD_ID>', data)

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
