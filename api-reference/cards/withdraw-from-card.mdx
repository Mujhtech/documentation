<RequestExample>

```bash cURL
curl -X POST https://api.swyftpay.io/api/v1/cards/{id}/withdraw \
  -H "Authorization: Bearer <TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{
    "amount": 1000
  }'
```

```javascript Node.js
import { SwervpayClient } from '@swervpaydev/sdk';

const config = {
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

const swyftpay = new SwervpayClient(config);

await swyftpay.card.withdraw('<CARD_ID>',{
    amount: "user"
})
```

```php PHP

```


```go Go
package main

import (
	"context"
	"fmt"
	swervpay "github.com/swerv-ltd/swervpay-go"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	client := swervpay.NewSwervpayClient(&swervpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })


    data := swervpay.CardWithdraw{
        Amount: 1000,
    }

	resp, err := client.Card.withdraw(ctx, '<CARD_ID>', data)

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
