```bash cURL

  -H "Authorization: Bearer <TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{
    "amount": 1000
  }'
```

```javascript Node.js
import { SwyftpayClient } from '@swervpaydev/sdk';

const config = {
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

const swyftpay = new SwyftpayClient(config);

await swyftpay.card.fund({
    cardId: "user",
    amount: "user"
})
```

```php PHP
use SwyftpayClient\SDK\Client;

$swyftpayClient = new Client(<SECRET_KEY>, <BUSINESS_ID>);

$swyftpayClient->card->fund([
    "cardId" => "user",
    "amount" => "user"
])->toArray();
```


```go Go
package main

import (
	"context"
	"fmt"
	swyftpay "github.com/Swerv-Ltd/swyftpay-go/lib"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	swyftpayClient := swyftpay.NewAClient(&swyftpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })


    data := swyftpayClient.CardFund{
        CardId: "user",
        Amount: 1000
    }

	resp, err := swyftpayClient.card.fund(ctx, data)

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
