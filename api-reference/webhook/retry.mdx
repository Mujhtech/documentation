```javascript Node.js
import { SwervpayClient } from '@swervpaydev/sdk';

const config = {
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

const swyftpay = new SwervpayClient(config);

await swyftpay.webhook.retry('<WEBHOOK_LOG_ID>')
```

```php PHP
use Swervpaydev\SDK\Swervpay;

$config = [
	'business_id' => '<BUSINESS_ID>',
	'secret_key' => '<SECRET_KEY>'
];

$client = new Swervpay($config);

$client->webhook->retry('<WEBHOOK_LOG_ID>');
```


```go Go
package main

import (
	"context"
	"fmt"
	swervpay "github.com/swerv-ltd/swervpay-go"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	client := swervpay.NewSwervpayClient(&swervpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })


	resp, err := client.Webhook.retry(ctx, "<WEBHOOK_LOG_ID>")

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
