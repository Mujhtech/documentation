```go Go
package main

import (
	"context"
	"fmt"
	swervpay "github.com/swerv-ltd/swervpay-go"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	client := swervpay.NewSwervpayClient(&swervpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })


    data := swervpay.CustomerCreate{
        Firstname: "user",
        Lastname: "user",
        Middlename: "user",
        Country: "user",
        Email: "user@mailinator.com"
    }

	resp, err := client.Customer.create(ctx, data)

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
