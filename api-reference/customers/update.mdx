```bash cURL
curl -X POST https://api.swyftpay.io/api/v1/customers/{id}/update \
  -H "Authorization: Bearer <TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{
    "phone_number": "user",
    "email": "user@mailinator.com"
  }'
```

```javascript Node.js
import { SwervpayClient } from '@swervpaydev/sdk';

const config = {
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

const swyftpay = new SwervpayClient(config);

await swyftpay.customer.update("<CUSTOMER_ID>", {
    phoneNumber: "user",
    email: "user@mailinator.com"
})
```

```php PHP
use Swervpaydev\SDK\Swervpay;

$config = [
	'business_id' => '<BUSINESS_ID>',
	'secret_key' => '<SECRET_KEY>'
];

$swyftpayClient = new Swervpay($config);

$swyftpayClient->customer->update("<CUSTOMER_ID>", [
    "phone_number" => "user",
    "email" => "user@mailinator.com"
])->toArray();
```


```go Go
package main

import (
	"context"
	"fmt"
	swyftpay "github.com/Swerv-Ltd/swyftpay-go/lib"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	swyftpayClient := swyftpay.NewAClient(&swyftpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })


    data := swyftpayClient.CustomerUpdate{
        PhoneNumber: "user",
        Email: "user@mailinator.com"
    }

	resp, err := swyftpayClient.customer.create(ctx, "<CUSTOMER_ID>", data)

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
