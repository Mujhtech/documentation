```bash cURL

  -H "Authorization: Bearer <TOKEN>" \
  -H "Content-Type: application/json" \
  -d '{
    "phone_number": "user",

  }'
```

```javascript Node.js
import { SwervpayClient } from '@swervpaydev/sdk';

const config = {
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

const swyftpay = new SwervpayClient(config);

await swyftpay.customer.update("<CUSTOMER_ID>", {
    phoneNumber: "user",
    email: "user@mailinator.com"
})
```

```php PHP
use Swervpaydev\SDK\Swervpay;

$config = [
	'business_id' => '<BUSINESS_ID>',
	'secret_key' => '<SECRET_KEY>'
];

$client = new Swervpay($config);

$client->customer->update("<CUSTOMER_ID>", [
    "phone_number" => "user",
    "email" => "user@mailinator.com"
])->toArray();
```


```go Go
package main

import (
	"context"
	"fmt"
	swervpay "github.com/swerv-ltd/swervpay-go"
	"log"
)

func main() {
	ctx := context.Background()

	client := swervpay.NewSwervpayClient(&swervpay.SwervpayClientOption{
        BusinessId: "<BUSINESS_ID>",
        SecretKey: "<SECRET_KEY>",
    })

    data := swervpay.UpdateCustomerBody{
        PhoneNumber: "user",
        Email: "user@mailinator.com"
    }

	resp, err := client.Customer.Update(ctx, "<CUSTOMER_ID>", data)

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
