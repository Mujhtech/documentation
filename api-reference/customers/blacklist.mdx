```javascript Node.js
import { SwervpayClient } from '@swervpaydev/sdk';

const config = {
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

const swyftpay = new SwervpayClient(config);

await swyftpay.customer.blacklist({
    customerId: "user"
})
```

```php PHP
use Swervpaydev\SDK\Swervpay;

$config = [
	'business_id' => '<BUSINESS_ID>',
	'secret_key' => '<SECRET_KEY>'
];

$client = new Swervpay($config);

$client->customer->blacklist('<CUSTOMER_ID>');
```


```go Go
package main

import (
	"context"
	"fmt"
	swervpay "github.com/swerv-ltd/swervpay-go"
	"log"
)

func main() {
	ctx := context.Background()

	client := swervpay.NewSwervpayClient(&swervpay.SwervpayClientOption{
        BusinessId: "<BUSINESS_ID>",
        SecretKey: "<SECRET_KEY>",
    })

	resp, err := client.Customer.Blacklist(ctx, "<CUSTOMER_ID>")

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
