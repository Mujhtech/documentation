```javascript Node.js
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

await swyftpay.customer.blacklist({

})
```



```go Go
package main

import (
	"context"
	"fmt"
	swyftpay "github.com/Swerv-Ltd/swyftpay-go/lib"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	swyftpayClient := swyftpay.NewAClient(&swyftpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })

    data := swyftpayClient.CustomerBlacklist{
        CustomerId: "user"
    }

	resp, err := swyftpayClient.customer.blacklist(ctx, data)

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
