```javascript Node.js

const config = {
    secretKey: "<SECRET_KEY>",
    businessId: "<BUSINESS_ID>"
}

await swyftpay.transaction.get('<TRANSACTION_ID>')
```

```php PHP

];

$swyftpayClient = new Swervpay($config);

$swyftpayClient->transaction->get('<TRANSACTION_ID>');
```


```go Go
package main

import (
	"context"
	"fmt"
	swyftpay "github.com/Swerv-Ltd/swyftpay-go/lib"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	swyftpayClient := swyftpay.NewAClient(&swyftpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })

	resp, err := swyftpayClient.transaction.get(ctx, '<TRANSACTION_ID>')

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
