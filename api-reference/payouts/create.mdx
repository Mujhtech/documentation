```go Go
package main

import (
	"context"
	"fmt"
	swyftpay "github.com/Swerv-Ltd/swyftpay-go/lib"
	"log"
)

func main() {
	secretKey := "<SECRET_KEY>"
    businessId := "<BUSINESS_ID>"
	ctx := context.Background()

	swyftpayClient := swyftpay.NewAClient(&swyftpay.Config{
        BusinessId: businessId
        SecretKey: secretKey
    })


    data := swyftpayClient.PayoutCreate{
        BankCode: "user",
        AccountNumber: "user",
        Amount: "user",
        Currency: "NGN",
        Reference: "user",
        Naration: "user",
        Email: "user@mailinator.com"
    }

	resp, err := swyftpayClient.payout.create(ctx, data)

	if err != nil {
		log.Fatal("error", err.Error())
		return
	}
}
```
